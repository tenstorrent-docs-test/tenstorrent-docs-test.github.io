25ALWI void tilize_init(uint32_t icb, uint32_t block, uint32_t ocb = 16)
26{
27 MATH(( llk_math_eltwise_unary_datacopy_init<A2D, BroadcastType::NONE, DST_ACCUM_MODE, false/*is_int_en*/, true/*tilize en*/>(false /*transpose of faces*/, false /*transpose within 16x16 face*/, icb) ));
28 MATH(( llk_math_pack_sync_init<DST_ACCUM_MODE>() ));
29 MATH(( llk_math_hw_configure_disaggregated() ));
30
31 PACK(( llk_pack_hw_configure_disaggregated<false, DST_ACCUM_MODE, ReluType::NO_RELU, 0, true/*tilize en*/>(ocb) ));
32 PACK(( llk_pack_init<false, false, true/*tilize en*/>(ocb) ));
33 PACK(( llk_setup_outputs() ));
34 PACK(( llk_pack_dest_init<false, DST_ACCUM_MODE>(ocb) ));
35
36 UNPACK(( llk_setup_operands() ));
37 UNPACK(( llk_unpack_tilize_hw_configure_disaggregated<DST_ACCUM_MODE>(icb) ));
38 UNPACK(( llk_unpack_tilize_init(icb, block) ));
39}

134ALWI void tilize_block(uint32_t icb, uint32_t block, uint32_t ocb)
135{
136 UNPACK(( llk_unpack_tilize_block(icb, block) ));
137
138 for (uint32_t t = 0; t < block; t++) {
139 // Acquire dst
140 MATH(( llk_math_wait_for_dest_available() ));
141 PACK(( llk_packer_wait_for_math_done() ));
142
143 // Datacopy
144 MATH(( llk_math_eltwise_unary_datacopy<A2D, BroadcastType::NONE>(0 /*dst index*/) ));
145 PACK(( llk_pack<false, false >(0 /*tile index*/, ocb) ));
146
147 // Release dest
148 MATH(( llk_math_dest_section_done<DST_ACCUM_MODE>() ));
149 PACK(( llk_pack_dest_section_done<DST_ACCUM_MODE>() ));
150 }
151}

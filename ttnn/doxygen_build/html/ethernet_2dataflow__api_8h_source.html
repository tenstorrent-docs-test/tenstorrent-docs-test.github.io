112FORCE_INLINE
113void eth_send_bytes(
114 uint32_t src_addr,
115 uint32_t dst_addr,
116 uint32_t num_bytes,
117 uint32_t num_bytes_per_send = 16,
118 uint32_t num_bytes_per_send_word_size = 1) {
119 uint32_t num_bytes_sent = 0;
120 while (num_bytes_sent < num_bytes) {
121 internal_::eth_send_packet(
122 0, ((num_bytes_sent + src_addr) >> 4), ((num_bytes_sent + dst_addr) >> 4), num_bytes_per_send_word_size);
123 num_bytes_sent += num_bytes_per_send;
124 }
125 erisc_info->channels[0].bytes_sent += num_bytes;
126}

371FORCE_INLINE
372void eth_wait_for_bytes_on_channel(uint32_t num_bytes, uint8_t channel) {
373 // assert(channel < 4);
374 uint32_t count = 0;
375 uint32_t poll_count = 1000000;
376 uint32_t num_bytes_sent = erisc_info->channels[channel].bytes_sent;
377 while (num_bytes_sent == 0) {
378 uint32_t received_this_iter = erisc_info->channels[channel].bytes_sent;
379 if (received_this_iter != num_bytes_sent) {
380 // We are currently in the process of receiving data on this channel, so we just just wait a
381 // bit longer instead of initiating a context switch
382 num_bytes_sent = received_this_iter;
383 } else {
384 count++;
385 if (count > poll_count) {
386 count = 0;
387 run_routing();
388 }
389 }
390 }
391}
